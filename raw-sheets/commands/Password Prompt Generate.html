set "psCommand=powershell -Command "$pword = read-host 'Enter Password' -AsSecureString ; ^

[System.Runtime.InteropServices.Marshal]::PtrToStringAuto($BSTR)""
for /f "usebackq delims=" %%p in (`%psCommand%`) do set password=%%p
